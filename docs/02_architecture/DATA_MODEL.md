## Domain Overview

The central entity is a **Run** — the complete lifecycle record of one engineering initiative. Everything else exists to support run creation, governance, execution, and learning.

```
Organisation
  └── Workspace
        └── Project
              └── Run
                    ├── PlanTask[]
                    ├── GateDecision[]
                    ├── ExecutionBatch[]
                    ├── AuditEvent[]
                    ├── CanonicalEvent[]
                    └── OutcomeRecord
```

## Primary Entities

### UserInput

The raw intent provided by the operator when creating a run.

```typescript
interface UserInput {
  idea: string;                         // Project idea / goal
  mode: Mode;                           // Execution mode
  skillLevel?: 'beginner' | 'intermediate' | 'advanced';
  priority?: 'speed' | 'quality' | 'cost';
  deliverable?: 'app' | 'api' | 'script' | 'report';
  projectId?: string;                   // Required for hosted multi-tenant flows
}
```

### Run

The top-level orchestration record.

```typescript
interface RunState {
  id: string;                           // run_YYYYMMDD_NNNN
  orgId: string;
  workspaceId?: string;
  projectId?: string;
  actorId: string;
  actorType: 'human' | 'service-account' | 'system';
  authMode: 'session' | 'service-account' | 'legacy-api-key';
  correlationId: string;
  idea: string;
  mode: Mode;
  status: RunStatus;
  priority?: string;
  deliverable?: string;
  createdAt: string;
  updatedAt: string;
}

type RunStatus = 'planned' | 'running' | 'paused' | 'completed' | 'failed' | 'cancelled';
```

### PlanTask

One discrete unit of work generated by the planner.

```typescript
interface PlanTask {
  id: string;
  runId: string;
  phase: Phase;                         // intake | planning | skills | gating | building | ...
  title: string;
  description: string;
  doneDefinition: string;
  position: number;                     // Ordering within the run
  status: StepStatus;
}

type Phase = 'intake' | 'planning' | 'skills' | 'gating' | 'building' | 'testing' | 'reviewing' | 'deployment';
type StepStatus = 'pending' | 'running' | 'success' | 'failed' | 'paused' | 'skipped' | 'rolled-back';
```

### GateDecision

A governance checkpoint decision for a run.

```typescript
interface GateDecision {
  id: string;
  runId: string;
  gateType: GateType;                   // All 14 gate types
  status: GateStatus;
  reason: string;
  shouldPause: boolean;
  decidedBy?: string;                   // actorId
  decidedAt?: string;                   // ISO timestamp
  decisionNote?: string;
}

type GateStatus = 'pass' | 'needs-review' | 'blocked' | 'pending' | 'approved' | 'rejected';
```

### AuditEvent

Immutable record of every material action — for governance, forensics, and non-repudiation.

```typescript
interface AuditEvent {
  id: string;
  runId?: string;
  orgId: string;
  workspaceId?: string;
  projectId?: string;
  actorId: string;
  actorType: string;
  authMode: string;
  correlationId: string;
  eventType: string;
  payload: Record<string, unknown>;
  createdAt: string;
}
```

### CanonicalEvent

Published events for realtime UI consumption and async processing.

```typescript
interface CanonicalEvent {
  id: string;
  runId?: string;
  eventName: string;                    // e.g. 'gate.awaiting_approval'
  payload: Record<string, unknown>;
  actorId?: string;
  actorType?: string;
  orgId?: string;
  workspaceId?: string;
  projectId?: string;
  authMode?: string;
  correlationId?: string;
  createdAt: string;
}
```

### OutcomeRecord

Post-run learning signal used by the learning engine.

```typescript
interface OutcomeRecord {
  id: string;
  runId: string;
  success: boolean;
  failures: FailureDetail[];
  retryCount: number;
  durationMs: number;
  qualityScore?: number;               // 0.0 – 1.0
  userFeedback?: string;
  operatorRating?: number;            // 1–5 (from ck outcome command)
  createdAt: string;
}
```

## PostgreSQL Schema (Canonical)

> Source file: `/db/schema.sql`
> Migrations: `/db/migrations/`

### Core Hierarchy Tables

```sql
-- organisations → workspaces → projects
CREATE TABLE organizations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE workspaces (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE projects (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  workspace_id UUID NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  slug TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(workspace_id, slug)
);
```

### Run and Task Tables

```sql
CREATE TABLE runs (
  id UUID PRIMARY KEY,
  organization_id UUID NOT NULL REFERENCES organizations(id),
  workspace_id UUID REFERENCES workspaces(id),
  project_id UUID REFERENCES projects(id),
  actor_id TEXT NOT NULL,
  actor_type TEXT NOT NULL,
  auth_mode TEXT NOT NULL,
  correlation_id TEXT NOT NULL,
  idea TEXT NOT NULL,
  mode TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'planned',
  priority TEXT,
  deliverable TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE plan_tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  run_id UUID NOT NULL REFERENCES runs(id) ON DELETE CASCADE,
  phase TEXT NOT NULL,
  title TEXT NOT NULL,
  description TEXT NOT NULL,
  done_definition TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'pending',
  position INT NOT NULL
);
```

### Governance Tables

```sql
CREATE TABLE gate_decisions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  run_id UUID NOT NULL REFERENCES runs(id) ON DELETE CASCADE,
  gate_type TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'pending',
  reason TEXT NOT NULL,
  should_pause BOOLEAN NOT NULL DEFAULT false,
  decided_by TEXT,
  decided_at TIMESTAMPTZ,
  decision_note TEXT
);

CREATE TABLE service_accounts (
  id TEXT PRIMARY KEY,
  name TEXT NOT NULL,
  org_id UUID NOT NULL REFERENCES organizations(id),
  workspace_id UUID REFERENCES workspaces(id),
  project_id UUID REFERENCES projects(id),
  scopes JSONB NOT NULL DEFAULT '[]'::jsonb,
  created_by TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

### Audit and Events Tables

```sql
CREATE TABLE audit_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  run_id UUID REFERENCES runs(id) ON DELETE SET NULL,
  organization_id UUID NOT NULL REFERENCES organizations(id),
  workspace_id UUID REFERENCES workspaces(id),
  project_id UUID REFERENCES projects(id),
  actor_id TEXT NOT NULL,
  actor_type TEXT NOT NULL,
  auth_mode TEXT NOT NULL,
  correlation_id TEXT NOT NULL,
  event_type TEXT NOT NULL,
  payload JSONB NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE canonical_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  run_id UUID REFERENCES runs(id) ON DELETE SET NULL,
  event_name TEXT NOT NULL,
  payload JSONB NOT NULL,
  actor_id TEXT,
  actor_type TEXT,
  org_id UUID REFERENCES organizations(id),
  workspace_id UUID REFERENCES workspaces(id),
  project_id UUID REFERENCES projects(id),
  auth_mode TEXT,
  correlation_id TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE outcome_records (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  run_id UUID NOT NULL REFERENCES runs(id) ON DELETE CASCADE,
  success BOOLEAN NOT NULL,
  failures JSONB NOT NULL DEFAULT '[]'::jsonb,
  retry_count INT NOT NULL DEFAULT 0,
  duration_ms INT NOT NULL,
  quality_score NUMERIC(5,4),
  user_feedback TEXT,
  operator_rating INT CHECK (operator_rating BETWEEN 1 AND 5),
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

## Indexing Strategy

```sql
-- Dashboard and CLI list queries
CREATE INDEX idx_runs_project_status ON runs(project_id, status, created_at DESC);
CREATE INDEX idx_runs_org ON runs(organization_id, created_at DESC);

-- Gate queries
CREATE INDEX idx_gate_decisions_run ON gate_decisions(run_id, status);
CREATE INDEX idx_gate_decisions_pending ON gate_decisions(status) WHERE status = 'pending';

-- Audit governance queries
CREATE INDEX idx_audit_events_org_type ON audit_events(organization_id, event_type, created_at DESC);
CREATE INDEX idx_audit_events_project ON audit_events(project_id, created_at DESC);
CREATE INDEX idx_audit_events_correlation ON audit_events(correlation_id);

-- Timeline rendering
CREATE INDEX idx_canonical_events_run ON canonical_events(run_id, created_at ASC);

-- Analytics
CREATE INDEX idx_outcome_records_success ON outcome_records(success, created_at DESC);
```

## Artifact Storage Metadata

Artifacts (reports, logs, previews) stored in InsForge or local filesystem use this metadata shape:

```typescript
interface ArtifactRecord {
  artifactId: string;
  runId: string;
  category: 'run-report' | 'log' | 'preview' | 'output';
  provider: 'local' | 'insforge-storage';
  path: string;
  contentType: string;
  sizeBytes: number;
  sha256: string;
  createdAt: string;
}
```
